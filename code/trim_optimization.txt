colnames(og) = gsub('\\.fastq.*','',gsub('.*_GEN','GEN',f))

# map vir3.fasta to VIR_clean
vir.fa = seqinr::read.fasta('../data-raw/peptide_table/vir3.fasta')
vir.csv= read.csv('../data-raw/peptide_table/VIR3_clean.csv')
mapping=lapply( vir.fa[1:1e4] ,
        function(x) which(grepl( paste(unlist(x),collapse='') , vir.csv$insert ) )) %>% melt()
colnames(mapping) = c('fa','csv')
save(mapping,file='../data-raw/peptide_table/VIR3_vir.mapping.rda')


load('../data-raw/peptide_table/VIR3_vir.mapping.rda')
og2=do.call(rbind, lapply( unique(mapping[,2]) , function(i) colSums(og[i,])/2 ))
rownames(og2) = as.character(unique(mapping[,2]))


sim=list()
for(i in names(new)){

        rn = intersect(rownames(og2),rownames(new[[i]]))
        ni = new[[i]][rownames(new[[i]])%in%rn,colnames(og2)] #%>% arrange('row.names')
        oi = og2[rownames(og2)%in%rn,colnames(og2)] #%>% arrange('row.names')

        if(any(rownames(ni)!=rownames(oi))){stop('mismatch aln')}

        sim[[i]]=sapply( colnames(ni),function(j){
                cor(sort(ni[,j]),sort(oi[,j]) ,method='spearman')
        })

}
tmp=do.call(rbind,sim)
head(tmp[order(-rowSums(tmp)),])
write.csv(tmp,'elledge_vs_new.csv')
